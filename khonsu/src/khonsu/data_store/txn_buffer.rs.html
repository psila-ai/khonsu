6/// The key type for data items in the store. Using String for now.
7///
8/// This type alias defines the type used for keys to identify data items
9/// within the `TxnBuffer`. Currently, it is set to `String`.
10type DataKey = String;
11
12/// The Transaction Buffer (TxnBuffer)
13///
14/// This struct represents the in-memory data store for the Software Transactional
15/// Memory (STM) system. It holds the current state of the data that active
16/// transactions operate on. The data is stored as a map of `DataKey` to
17/// `VersionedValue`, protected by a `parking_lot::RwLock` for concurrent access.
18pub struct TxnBuffer {
19    data: RwLock<HashMap<DataKey, VersionedValue>>,
20}
21
22impl Default for TxnBuffer {
23    /// Creates a new, empty `TxnBuffer` with default settings.
24    fn default() -> Self {
25        Self::new()
26    }
27}
28
29impl TxnBuffer {
30    /// Creates a new, empty `TxnBuffer`.
31    ///
32    /// Initializes an empty in-memory data buffer protected by a reader-writer lock.
33    ///
34    /// # Returns
35    ///
36    /// A new, empty `TxnBuffer` instance.
37    ///
38    /// # Examples
39    ///
40    /// ```no_run
41    /// use khonsu::prelude::*;
42    ///
43    /// let txn_buffer = TxnBuffer::new();
44    /// ```
45    pub fn new() -> Self {
46        Self {
47            data: RwLock::new(HashMap::new()),
48        }
49    }
50
51    /// Retrieves a `VersionedValue` from the buffer based on the key.
52    ///
53    /// This method provides read access to the data in the transaction buffer.
54    /// It acquires a read lock on the internal data map.
55    ///
56    /// # Arguments
57    ///
58    /// * `key` - A reference to the `DataKey` of the data item to retrieve.
59    ///
60    /// # Returns
61    ///
62    /// Returns `Some(VersionedValue)` if the key exists in the buffer,
63    /// otherwise returns `None`. The returned `VersionedValue` is cloned.
64    ///
65    /// # Examples
66    ///
67    /// ```no_run
68    /// use khonsu::prelude::*;
69    /// use std::sync::Arc;
70    /// use arrow::record_batch::RecordBatch;
71    /// use arrow::array::Int32Array;
72    /// use arrow::datatypes::{Schema, Field, DataType};
73    ///
74    /// # let txn_buffer = TxnBuffer::new();
75    /// let key = "my_data".to_string();
76    ///
77    /// // Assume some data has been inserted into the buffer.
78    /// // let schema = Arc::new(Schema::new(vec![Field::new("value", DataType::Int32, false)]));
79    /// // let value_array = Int32Array::from(vec![10]);
80    /// // let record_batch = RecordBatch::try_new(schema, vec![Arc::new(value_array)]).unwrap();
81    /// // let versioned_value = VersionedValue::new(Arc::new(record_batch), 1);
82    /// // txn_buffer.insert(key.clone(), versioned_value);
83    ///
84    /// match txn_buffer.get(&key) {
85    ///     Some(versioned_value) => {
86    ///         println!("Retrieved data for key {}: {:?}", key, versioned_value.data());
87    ///     }
88    ///     None => {
89    ///         println!("Key {} not found in buffer.", key);
90    ///     }
91    /// }
92    /// ```
93    pub fn get(&self, key: &DataKey) -> Option<VersionedValue> {
94        let data = self.data.read();
95        data.get(key).cloned()
96    }
97
98    /// Inserts or updates a `VersionedValue` in the buffer.
99    ///
100    /// This method inserts a new data item or updates an existing one with the
101    /// provided `VersionedValue`. It acquires a write lock on the internal
102    /// data map to ensure exclusive access during the modification.
103    ///
104    /// # Arguments
105    ///
106    /// * `key` - The `DataKey` for the data item.
107    /// * `value` - The `VersionedValue` to insert or update.
108    ///
109    /// # Returns
110    ///
111    /// Returns `Some(VersionedValue)` containing the previous value associated
112    /// with the key if it existed, otherwise returns `None`.
113    ///
114    /// # Examples
115    ///
116    /// ```no_run
117    /// use khonsu::prelude::*;
118    /// use std::sync::Arc;
119    /// use arrow::record_batch::RecordBatch;
120    /// use arrow::array::Int32Array;
121    /// use arrow::datatypes::{Schema, Field, DataType};
122    ///
123    /// # let txn_buffer = TxnBuffer::new();
124    /// let key = "new_data".to_string();
125    /// let schema = Arc::new(Schema::new(vec![Field::new("value", DataType::Int32, false)]));
126    /// let value_array = Int32Array::from(vec![100]);
127    /// let record_batch = RecordBatch::try_new(schema, vec![Arc::new(value_array)]).unwrap();
128    /// let versioned_value = VersionedValue::new(Arc::new(record_batch), 1);
129    ///
130    /// match txn_buffer.insert(key.clone(), versioned_value) {
131    ///     Some(previous_value) => {
132    ///         println!("Updated data for key {}. Previous version: {}", key, previous_value.version());
133    ///     }
134    ///     None => {
135    ///         println!("Inserted data for key {}.", key);
136    ///     }
137    /// }
138    /// ```
139    pub fn insert(&self, key: DataKey, value: VersionedValue) -> Option<VersionedValue> {
140        let mut data = self.data.write();
141        data.insert(key, value)
142    }
143
144    /// Removes a value from the buffer based on the key.
145    ///
146    /// This method removes the data item associated with the given `key` from
147    /// the transaction buffer. It acquires a write lock on the internal data map.
148    ///
149    /// # Arguments
150    ///
151    /// * `key` - A reference to the `DataKey` of the data item to remove.
152    ///
153    /// # Returns
154    ///
155    /// Returns `Some(VersionedValue)` containing the removed value if the key
156    /// existed in the buffer, otherwise returns `None`.
157    ///
158    /// # Examples
159    ///
160    /// ```no_run
161    /// use khonsu::prelude::*;
162    /// use std::sync::Arc;
163    /// use arrow::record_batch::RecordBatch;
164    /// use arrow::array::Int32Array;
165    /// use arrow::datatypes::{Schema, Field, DataType};
166    ///
167    /// # let txn_buffer = TxnBuffer::new();
168    /// let key = "data_to_delete".to_string();
169    ///
170    /// // Assume some data exists in the buffer for this key.
171    /// // let schema = Arc::new(Schema::new(vec![Field::new("value", DataType::Int32, false)]));
172    /// // let value_array = Int32Array::from(vec![50]);
173    /// // let record_batch = RecordBatch::try_new(schema, vec![Arc::new(value_array)]).unwrap();
174    /// // let versioned_value = VersionedValue::new(Arc::new(record_batch), 2);
175    /// // txn_buffer.insert(key.clone(), versioned_value);
176    ///
177    /// match txn_buffer.delete(&key) {
178    ///     Some(removed_value) => {
179    ///         println!("Removed data for key {}. Removed version: {}", key, removed_value.version());
180    ///     }
181    ///     None => {
182    ///         println!("Key {} not found in buffer, nothing to delete.", key);
183    ///     }
184    /// }
185    /// ```
186    pub fn delete(&self, key: &DataKey) -> Option<VersionedValue> {
187        let mut data = self.data.write();
188        data.remove(key)
189    }
