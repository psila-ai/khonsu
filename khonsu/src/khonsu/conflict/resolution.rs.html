1/// Defines how conflicts should be resolved during transaction commit.
2///
3/// When a transaction attempts to commit and conflicts are detected with
4/// concurrently committed transactions, the configured `ConflictResolution`
5/// strategy determines how the system should proceed.
6#[derive(Debug, Clone, Copy, PartialEq, Eq)]
7pub enum ConflictResolution {
8    /// Append the new data from the committing transaction to the existing data
9    /// in the transaction buffer for the conflicting key. This strategy is
10    /// typically used for append-only data structures or logs.
11    Append,
12    /// Ignore the conflicting change from the committing transaction. The data
13    /// in the transaction buffer remains unchanged for the conflicting key.
14    Ignore,
15    /// Replace the existing data in the transaction buffer with the new data
16    /// from the committing transaction for the conflicting key.
17    Replace,
18    /// Fail the transaction if a conflict is detected. The transaction is
19    /// aborted, and none of its changes are applied.
20    Fail,
21}
