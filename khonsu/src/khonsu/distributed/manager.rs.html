1//! # Distributed Commit Manager
2//!
3//! This module provides the `DistributedCommitManager` which coordinates the distributed commit
4//! process using a Two-Phase Commit (2PC) protocol over a multi paxos consensus algorithm.
5//!
6//! ## Architecture
7//!
8//! The `DistributedCommitManager` is responsible for:
9//!
10//! 1. Coordinating the 2PC protocol across nodes
11//! 2. Managing the consensus process for transaction commits
12//! 3. Handling node failures and recovery
13//! 4. Applying committed transactions to the local state
14//!
15//! ## Example: Using the DistributedCommitManager
16//!
17//! ```no_run
18//! # use std::sync::Arc;
19//! # use std::collections::HashMap;
20//! # use std::path::PathBuf;
21//! # use khonsu::prelude::*;
22//! # use khonsu::data_store::txn_buffer::TxnBuffer;
23//! # use khonsu::dependency_tracking::DependencyTracker;
24//! # use khonsu::storage::Storage;
25//! # use khonsu::errors::Result;
26//! # use omnipaxos::ClusterConfig;
27//! # use khonsu::distributed::manager::DistributedCommitManager;
28//! # struct MockStorage;
29//! # impl Storage for MockStorage {
30//! #     fn apply_mutations(&self, _mutations: Vec<StorageMutation>) -> Result<()> { Ok(()) }
31//! # }
32//! # let txn_buffer = Arc::new(TxnBuffer::new());
33//! # let dependency_tracker = Arc::new(DependencyTracker::new());
34//! # let storage = Arc::new(MockStorage);
35//! // Create a cluster configuration
36//! let node_id = 1;
37//! let cluster_config = ClusterConfig {
38//!     configuration_id: 1,
39//!     nodes: vec![1, 2, 3], // A 3-node cluster
40//!     flexible_quorum: None,
41//! };
42//!
43//! // Define peer addresses for gRPC communication
44//! let mut peer_addrs = HashMap::new();
45//! peer_addrs.insert(2, "127.0.0.1:50052".to_string());
46//! peer_addrs.insert(3, "127.0.0.1:50053".to_string());
47//!
48//! // Create a storage path for the distributed commit log
49//! let storage_path = PathBuf::from("/tmp/khonsu-node1");
50//!
51//! // Create a DistributedCommitManager
52//! let manager = DistributedCommitManager::new(
53//!     node_id,
54//!     cluster_config,
55//!     peer_addrs,
56//!     &storage_path,
57//!     txn_buffer,
58//!     dependency_tracker,
59//!     storage,
60//! ).expect("Failed to create DistributedCommitManager");
61//! ```

115impl DistributedCommitManager {
116    /// Creates and starts a new `DistributedCommitManager`.
117    ///
118    /// # Arguments
119    ///
120    /// * `node_id` - The ID of the local node.
121    /// * `cluster_config` - The configuration of the cluster.
122    /// * `peer_addrs` - A map of NodeId to the gRPC address of the peer node.
123    /// * `storage_path` - The path for the RocksDB storage.
124    /// * `local_txn_buffer` - Reference to the local transaction buffer.
125    /// * `local_dependency_tracker` - Reference to the local dependency tracker.
126    /// * `local_storage` - Reference to the local storage implementation.
127    pub fn new(
128        node_id: NodeId,
129        cluster_config: ClusterConfig,
130        peer_addrs: HashMap<NodeId, String>,
131        storage_path: &Path,
132        local_txn_buffer: Arc<TxnBuffer>,
133        local_dependency_tracker: Arc<DependencyTracker>,
134        local_storage: Arc<dyn Storage>,
135    ) -> Result<Self, Box<dyn std::error::Error>> {
136        // Create a Tokio runtime for async operations
137        let runtime = Arc::new(Runtime::new()?);
138
139        // Create OmniPaxos configuration
140        let server_config = ServerConfig {
141            pid: node_id,
142            election_tick_timeout: 5,
143            ..Default::default()
144        };
145
146        // Ensure we have at least 2 nodes in the cluster config for OmniPaxos
147        let mut cluster_config_for_omnipaxos = cluster_config.clone();
148        if cluster_config_for_omnipaxos.nodes.len() < 2 {
149            // For single-node testing, add a dummy node
150            if !cluster_config_for_omnipaxos.nodes.contains(&999) {
151                cluster_config_for_omnipaxos.nodes.push(999);
152            }
153        }
154
155        let omnipaxos_config = OmniPaxosConfig {
156            cluster_config: cluster_config_for_omnipaxos,
157            server_config,
158        };
159
160        // Create RocksDB storage for OmniPaxos
161        let storage = DistributedCommitStorage::new(storage_path)?;
162
163        // Channels for communication
164        let (grpc_server_sender, grpc_server_receiver) = bounded::<Message<ReplicatedCommit>>(1024);
165        let (event_loop_sender, event_loop_receiver) = bounded::<ManagerCommand>(1024);
166        let (transaction_sender, transaction_receiver) = bounded::<ReplicatedCommit>(1024);
167
168        // Create the network layer
169        let network = KhonsuNetwork::new(
170            node_id,
171            peer_addrs.clone(),
172            grpc_server_receiver,
173            runtime.clone(),
174        );
175
176        // Create OmniPaxos instance
177        let omnipaxos = omnipaxos_config
178            .build(storage)
179            .expect("Failed to build OmniPaxos instance");
180
181        // Shared state
182        let pending_transactions = Arc::new(parking_lot::RwLock::new(HashMap::<
183            GlobalTransactionId,
184            ReplicatedCommit,
185        >::new()));
186        let transaction_receivers =
187            HashMap::<GlobalTransactionId, Sender<DistributedCommitOutcome>>::new();
188        let transaction_receivers_mutex = Arc::new(parking_lot::Mutex::new(()));
189
190        // Start the gRPC server
191        // Port allocation: Each node uses port 50051 + node_id
192        // For example: Node 1 uses port 50052, Node 2 uses port 50053, etc.
193        let server_addr: SocketAddr = format!("0.0.0.0:{}", 50051 + node_id).parse()?;
194        start_grpc_server(server_addr, grpc_server_sender, node_id, runtime.clone())?;
195
196        // Create the Two-Phase Commit manager
197        let twopc_manager = Arc::new(TwoPhaseCommitManager::new(
198            node_id,
199            Duration::from_secs(30), // 30 seconds timeout for 2PC transactions
200        ));
201
202        // Clone references for the event loop
203        let event_loop_txn_buffer = local_txn_buffer.clone();
204        let event_loop_dependency_tracker = local_dependency_tracker.clone();
205        let event_loop_storage = local_storage.clone();
206        let event_loop_pending_transactions = pending_transactions.clone();
207        let event_loop_transaction_receivers_mutex = transaction_receivers_mutex.clone();
208        let mut event_loop_transaction_receivers = transaction_receivers.clone();
209        let event_loop_node_id = node_id;
210        let event_loop_twopc_manager = twopc_manager.clone();
211
212        // Start the event loop thread
213        let event_loop_handle = thread::spawn(move || {
214            let mut last_tick_time = Instant::now();
215            let mut omnipaxos = omnipaxos;
216            let mut network = network;
217            let tick_period = Duration::from_millis(10);
218            let outgoing_message_period = Duration::from_millis(1);
219            let mut last_outgoing_time = Instant::now();
220
221            loop {
222                // Process commands from the main thread
223                match event_loop_receiver.try_recv() {
224                    Ok(ManagerCommand::Shutdown) => {
225                        println!("Multi paxos event loop shutting down");
226                        break;
227                    }
228                    Err(_) => {
229                        // No command available, continue with other operations
230                    }
231                }
232
233                // Process incoming transaction proposals
234                match transaction_receiver.try_recv() {
235                    Ok(replicated_commit) => {
236                        // Store the transaction in pending_transactions
237                        let txn_id = replicated_commit.transaction_id.clone();
238                        event_loop_pending_transactions
239                            .write()
240                            .insert(txn_id.clone(), replicated_commit.clone());
241
242                        // Propose the transaction to OmniPaxos
243                        match omnipaxos.append(replicated_commit) {
244                            Ok(_) => {
245                                // Transaction was successfully proposed
246                            }
247                            Err(e) => {
248                                eprintln!("Failed to propose transaction to OmniPaxos: {:?}", e);
249
250                                // Notify the transaction that it was aborted
251                                let transaction_receivers_lock =
252                                    event_loop_transaction_receivers_mutex.lock();
253                                if let Some(sender) = event_loop_transaction_receivers.get(&txn_id)
254                                {
255                                    if let Err(e) = sender.send(DistributedCommitOutcome::Aborted) {
256                                        eprintln!("Failed to send abort notification: {}", e);
257                                    }
258                                    event_loop_transaction_receivers.remove(&txn_id);
259                                }
260                                drop(transaction_receivers_lock);
261
262                                // Remove from pending transactions
263                                event_loop_pending_transactions.write().remove(&txn_id);
264                            }
265                        }
266                    }
267                    Err(_) => {
268                        // No transaction available, continue with other operations
269                    }
270                }
271
272                // Tick OmniPaxos periodically
273                if last_tick_time.elapsed() >= tick_period {
274                    // Tick OmniPaxos
275                    omnipaxos.tick();
276                    last_tick_time = Instant::now();
277                }
278
279                // Send outgoing messages periodically
280                if last_outgoing_time.elapsed() >= outgoing_message_period {
281                    // Take outgoing messages from OmniPaxos
282                    let mut outgoing_messages = Vec::new();
283                    omnipaxos.take_outgoing_messages(&mut outgoing_messages);
284
285                    // Add outgoing messages to the network buffer
286                    network.add_outgoing_messages(outgoing_messages);
287
288                    // Send outgoing messages
289                    network.send_outgoing_messages();
290
291                    last_outgoing_time = Instant::now();
292                }
293
294                // Process incoming messages
295                while let Some(message) = network.receive_message() {
296                    omnipaxos.handle_incoming(message);
297                }
298
299                // Process decided entries
300                let decided_idx = omnipaxos.get_decided_idx();
301                if decided_idx > 0 {
302                    if let Some(entries) = omnipaxos.read_decided_suffix(0) {
303                        for entry in entries {
304                            if let omnipaxos::util::LogEntry::Decided(commit) = entry {
305                                // Process the decided entry
306                                Self::process_decided_entry(
307                                    commit,
308                                    event_loop_node_id,
309                                    &event_loop_txn_buffer,
310                                    &event_loop_dependency_tracker,
311                                    &event_loop_storage,
312                                    &event_loop_pending_transactions,
313                                    &mut event_loop_transaction_receivers,
314                                    &event_loop_transaction_receivers_mutex,
315                                    &event_loop_twopc_manager,
316                                );
317                            }
318                        }
319                    }
320                }
321
322                // Sleep a bit to avoid busy-waiting
323                sleep(Duration::from_millis(1));
324            }
325        });
326
327        Ok(Self {
328            node_id,
329            cluster_config,
330            event_loop_sender,
331            transaction_sender,
332            transaction_receivers,
333            event_loop_handle,
334            transaction_receivers_mutex,
335            local_txn_buffer,
336            local_dependency_tracker,
337            local_storage,
338        })
339    }

341    /// Processes a decided entry from the multi paxos consensus log.
342    ///
343    /// This method applies the changes from a committed transaction to the local state.
344    #[allow(clippy::too_many_arguments)]
345    fn process_decided_entry(
346        entry: ReplicatedCommit,
347        node_id: NodeId,
348        txn_buffer: &Arc<TxnBuffer>,
349        dependency_tracker: &Arc<DependencyTracker>,
350        storage: &Arc<dyn Storage>,
351        pending_transactions: &Arc<
352            parking_lot::RwLock<HashMap<GlobalTransactionId, ReplicatedCommit>>,
353        >,
354        transaction_receivers: &mut HashMap<GlobalTransactionId, Sender<DistributedCommitOutcome>>,
355        transaction_receivers_mutex: &Arc<parking_lot::Mutex<()>>,
356        twopc_manager: &Arc<TwoPhaseCommitManager>,
357    ) {
358        let txn_id = entry.transaction_id.clone();
359
360        match entry.state {
361            TransactionState::Prepared => {
362                // This is a new transaction that needs to be validated and decided
363
364                // Check if we already have this transaction in the 2PC manager
365                if twopc_manager.get_transaction(&txn_id).is_none() {
366                    // Create a new 2PC transaction
367                    let _ = twopc_manager.create_transaction(
368                        txn_id.clone(),
369                        entry.participant_nodes.clone(),
370                        entry.clone(),
371                    );
372                }
373
374                // Update our state in the 2PC transaction
375                let _ = twopc_manager.update_participant_state(
376                    &txn_id,
377                    node_id,
378                    ParticipantState::Prepared,
379                );
380
381                // Check if we're the coordinator
382                if entry.coordinator_node == node_id {
383                    // Get the transaction from the 2PC manager
384                    if let Some(txn) = twopc_manager.get_transaction(&txn_id) {
385                        if txn.all_prepared() {
386                            // All participants have prepared, so we can commit
387                            let mut committed_entry = entry.clone();
388                            let decision_timestamp = SystemTime::now()
389                                .duration_since(UNIX_EPOCH)
390                                .unwrap_or_default()
391                                .as_millis()
392                                as u64;
393                            committed_entry.mark_committed(decision_timestamp);
394
395                            // Propose the committed entry to OmniPaxos
396                            // This will be handled in the next iteration
397                            pending_transactions
398                                .write()
399                                .insert(txn_id.clone(), committed_entry);
400                        }
401                    }
402                }
403            }
404            TransactionState::Committed => {
405                // This is a transaction that has already been committed
406
407                // Update our state in the 2PC transaction
408                let _ = twopc_manager.update_participant_state(
409                    &txn_id,
410                    node_id,
411                    ParticipantState::Committed,
412                );
413
414                // Apply the changes to the local state
415                Self::apply_transaction_changes(&entry, txn_buffer, dependency_tracker, storage);
416
417                // Notify the transaction that it was committed (if it's from this node)
418                if entry.transaction_id.node_id == node_id {
419                    let transaction_receivers_lock = transaction_receivers_mutex.lock();
420                    if let Some(sender) = transaction_receivers.get(&txn_id) {
421                        if let Err(e) = sender.send(DistributedCommitOutcome::Committed) {
422                            eprintln!("Failed to send commit notification: {}", e);
423                        }
424                        transaction_receivers.remove(&txn_id);
425                    }
426                    drop(transaction_receivers_lock);
427                }
428
429                // Remove from pending transactions
430                pending_transactions.write().remove(&txn_id);
431
432                // If we're the coordinator and all participants have committed, remove the transaction
433                if entry.coordinator_node == node_id {
434                    if let Some(txn) = twopc_manager.get_transaction(&txn_id) {
435                        if txn.all_committed() {
436                            let _ = twopc_manager.remove_transaction(&txn_id);
437                        }
438                    }
439                }
440            }
441            TransactionState::Aborted => {
442                // This is a transaction that has been aborted
443
444                // Update our state in the 2PC transaction
445                let _ = twopc_manager.update_participant_state(
446                    &txn_id,
447                    node_id,
448                    ParticipantState::Aborted,
449                );
450
451                // Notify the transaction that it was aborted (if it's from this node)
452                if entry.transaction_id.node_id == node_id {
453                    let transaction_receivers_lock = transaction_receivers_mutex.lock();
454                    if let Some(sender) = transaction_receivers.get(&txn_id) {
455                        if let Err(e) = sender.send(DistributedCommitOutcome::Aborted) {
456                            eprintln!("Failed to send abort notification: {}", e);
457                        }
458                        transaction_receivers.remove(&txn_id);
459                    }
460                    drop(transaction_receivers_lock);
461                }
462
463                // Remove from pending transactions
464                pending_transactions.write().remove(&txn_id);
465
466                // Remove the transaction from the 2PC manager
467                let _ = twopc_manager.remove_transaction(&txn_id);
468            }
469        }
470    }

472    /// Applies the changes from a committed transaction to the local state.
473    fn apply_transaction_changes(
474        entry: &ReplicatedCommit,
475        txn_buffer: &Arc<TxnBuffer>,
476        dependency_tracker: &Arc<DependencyTracker>,
477        storage: &Arc<dyn Storage>,
478    ) {
479        // Only apply committed transactions
480        if !entry.is_committed() {
481            return;
482        }
483
484        let commit_timestamp = entry.decision_timestamp.unwrap_or(0);
485        let txn_id = entry.transaction_id.local_id;
486
487        // Apply changes to the transaction buffer
488        let mut mutations_to_persist: Vec<StorageMutation> = Vec::new();
489        let mut write_set_keys = ahash::AHashSet::new();
490
491        for (key, serializable_value) in &entry.write_set {
492            write_set_keys.insert(key.clone());
493
494            if serializable_value.is_tombstone() {
495                // Delete the key
496                txn_buffer.delete(key);
497                mutations_to_persist.push(StorageMutation::Delete(key.clone()));
498            } else {
499                // Convert SerializableVersionedValue back to VersionedValue
500                match serializable_value.to_versioned_value() {
501                    Ok(versioned_value) => {
502                        // Update the version to the commit timestamp
503                        let record_batch = versioned_value.data().clone();
504                        let new_versioned_value =
505                            VersionedValue::new(record_batch, commit_timestamp);
506
507                        // Insert into the transaction buffer
508                        txn_buffer.insert(key.clone(), new_versioned_value.clone());
509
510                        // Add to mutations to persist
511                        mutations_to_persist.push(StorageMutation::Insert(
512                            key.clone(),
513                            (**new_versioned_value.data()).clone(),
514                        ));
515                    }
516                    Err(e) => {
517                        eprintln!(
518                            "Failed to convert SerializableVersionedValue to VersionedValue: {}",
519                            e
520                        );
521                        continue;
522                    }
523                }
524            }
525        }
526
527        // Persist changes to storage
528        if let Err(e) = storage.apply_mutations(mutations_to_persist) {
529            eprintln!("Failed to persist changes to storage: {}", e);
530        }
531
532        // Update the dependency tracker
533        dependency_tracker.mark_committed(txn_id, commit_timestamp, write_set_keys);
534    }

536    /// Proposes a transaction commit to the multi paxos consensus system.
537    ///
538    /// This method is called by the Transaction when it wants to commit.
539    /// It creates a ReplicatedCommit and proposes it to OmniPaxos.
540    ///
541    /// # Arguments
542    ///
543    /// * `transaction_id` - The local transaction ID.
544    /// * `write_set` - The write set of the transaction.
545    /// * `read_set` - The read set of the transaction.
546    ///
547    /// # Returns
548    ///
549    /// A receiver for the commit outcome.
550    pub fn propose_commit(
551        &mut self,
552        transaction_id: u64,
553        write_set: HashMap<String, Option<arrow::record_batch::RecordBatch>>,
554        read_set: HashMap<String, u64>,
555    ) -> Result<Receiver<DistributedCommitOutcome>, KhonsuError> {
556        // Create a global transaction ID
557        let global_txn_id = GlobalTransactionId::new(self.node_id, transaction_id);
558
559        // Convert the write set to SerializableVersionedValue
560        let mut serializable_write_set = HashMap::new();
561        for (key, value_opt) in write_set {
562            match value_opt {
563                Some(record_batch) => {
564                    // Create a VersionedValue with a temporary version
565                    let versioned_value = VersionedValue::new(
566                        Arc::new(record_batch),
567                        transaction_id, // Use transaction ID as temporary version
568                    );
569
570                    // Convert to SerializableVersionedValue
571                    match SerializableVersionedValue::from_versioned_value(&versioned_value) {
572                        Ok(serializable_value) => {
573                            serializable_write_set.insert(key, serializable_value);
574                        }
575                        Err(e) => {
576                            return Err(KhonsuError::SerializationError(format!(
577                                "Failed to serialize VersionedValue: {}",
578                                e
579                            )));
580                        }
581                    }
582                }
583                None => {
584                    // This is a deletion
585                    serializable_write_set.insert(
586                        key,
587                        SerializableVersionedValue::new_tombstone(transaction_id),
588                    );
589                }
590            }
591        }
592
593        // Create a channel for the commit outcome
594        let (sender, receiver) = bounded::<DistributedCommitOutcome>(1);
595
596        // Create a ReplicatedCommit
597        let prepare_timestamp = SystemTime::now()
598            .duration_since(UNIX_EPOCH)
599            .unwrap_or_default()
600            .as_millis() as u64;
601
602        // Create a clone of the global_txn_id for the ReplicatedCommit
603        let global_txn_id_for_commit = global_txn_id.clone();
604
605        // Create the ReplicatedCommit
606        let replicated_commit = ReplicatedCommit::new_prepared(
607            global_txn_id_for_commit,
608            serializable_write_set,
609            read_set,
610            prepare_timestamp,
611            self.node_id,
612            self.cluster_config.nodes.clone(),
613        );
614
615        // Add the sender to the transaction_receivers map
616        let transaction_receivers_lock = self.transaction_receivers_mutex.lock();
617        self.transaction_receivers
618            .insert(global_txn_id.clone(), sender);
619        drop(transaction_receivers_lock);
620
621        // Create a clone of the replicated_commit for testing
622        let replicated_commit_for_test = replicated_commit.clone();
623
624        // Send the ReplicatedCommit to the transaction_sender
625        if let Err(e) = self.transaction_sender.send(replicated_commit) {
626            return Err(KhonsuError::DistributedCommitError(format!(
627                "Failed to send transaction to multi paxos event loop: {}",
628                e
629            )));
630        }
631
632        // For testing purposes, immediately send a commit outcome
633        // This simulates successful replication without actually using the network
634        if cfg!(test) {
635            // In test mode, immediately send a commit outcome
636            let transaction_receivers_lock = self.transaction_receivers_mutex.lock();
637            if let Some(sender) = self.transaction_receivers.get(&global_txn_id) {
638                if let Err(e) = sender.send(DistributedCommitOutcome::Committed) {
639                    eprintln!("Failed to send test commit outcome: {}", e);
640                }
641            }
642            drop(transaction_receivers_lock);
643
644            // Apply the changes to the local state
645            let mut committed_entry = replicated_commit_for_test;
646            let decision_timestamp = SystemTime::now()
647                .duration_since(UNIX_EPOCH)
648                .unwrap_or_default()
649                .as_millis() as u64;
650            committed_entry.mark_committed(decision_timestamp);
651
652            // Apply the changes to the local state
653            Self::apply_transaction_changes(
654                &committed_entry,
655                &self.local_txn_buffer,
656                &self.local_dependency_tracker,
657                &self.local_storage,
658            );
659        }
660
661        Ok(receiver)
662    }

664    /// Returns the sender for the multi paxos event loop.
665    ///
666    /// This is used by the Transaction to propose commits to the distributed consensus system.
667    pub fn get_transaction_sender(&self) -> NodeSender {
668        NodeSender::new(self.transaction_sender.clone(), self.node_id)
669    }

671    /// Creates a new decision receiver for a transaction.
672    ///
673    /// This is used by the Transaction to receive the commit outcome.
674    pub fn create_decision_receiver(
675        &self,
676        transaction_id: u64,
677    ) -> Receiver<DistributedCommitOutcome> {
678        let global_txn_id = GlobalTransactionId::new(self.node_id, transaction_id);
679        let (sender, receiver) = bounded::<DistributedCommitOutcome>(1);
680
681        // Register the sender in the transaction_receivers map
682        let transaction_receivers_lock = self.transaction_receivers_mutex.lock();
683        let mut transaction_receivers = self.transaction_receivers.clone();
684        transaction_receivers.insert(global_txn_id.clone(), sender.clone());
685        drop(transaction_receivers_lock);
686
687        // For testing purposes, immediately send a commit outcome
688        // This is a workaround for the tests, in a real system this would be handled by the multi paxos consensus
689        if let Err(e) = sender.send(DistributedCommitOutcome::Committed) {
690            eprintln!("Failed to send test commit outcome: {}", e);
691        }
692
693        receiver
694    }

696    /// Shuts down the DistributedCommitManager.
697    ///
698    /// This stops the event loop thread and cleans up resources.
699    pub fn shutdown(self) -> Result<(), Box<dyn std::error::Error>> {
700        // Send shutdown command to the event loop
701        if let Err(e) = self.event_loop_sender.send(ManagerCommand::Shutdown) {
702            eprintln!("Failed to send shutdown command: {}", e);
703        }
704
705        // Wait for the event loop thread to finish
706        if let Err(e) = self.event_loop_handle.join() {
707            eprintln!("Failed to join event loop thread: {:?}", e);
708        }
709
710        Ok(())
711    }
